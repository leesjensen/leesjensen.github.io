Filter

You can chain filters in a template to modify the output. Custom filters are created on the Vue instance.

Result: {{message | reverse | capitalize}} 

<div id="app">
  <p>Result: {{message | reverse | capitalize}}</p>
  <input v-model="message" />
</div>

<script>
  Vue.filter('capitalize', function (value) {
    return value.toUpperCase();
  });

  var app = new Vue({
    el: '#app',
    data: { message: '' },
    filters: {
      reverse: function (value) {
        return value.split('').reverse().join('');
      },
    },
  });
</script>
