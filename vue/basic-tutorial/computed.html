Computed

Computed properties represent the result of complex computations. Vue will cache the computed property and only update it if the property's dependencies change.

Date: {{date}}

Locale:

{{localizedDate}}

<div id="app">
  <p>Date: {{date}}</p>
  <p>
    Locale:
    <template v-for="(val, key) in locales">
      <input type="radio" :id="key" :value="val" v-model="localeCode" />
      <label v-bind:for="key">{{key}}</label>
    </template>
  </p>
  <p><span class="hlb">{{localizedDate}}</span></p>
</div>

<script>
  const standardLocales = {
    English: 'en-US',
    German: 'de-DE',
    Arabic: 'ar-EG',
    Japanese: 'ja-JP-u-ca-japanese',
  };
  var app = new Vue({
    el: '#app',
    data: {
      date: new Date(),
      locales: standardLocales,
      localeCode: 'de-DE',
    },
    computed: {
      localizedDate: function () {
        return this.date.toLocaleString(this.localeCode);
      },
    },
  });
</script>
