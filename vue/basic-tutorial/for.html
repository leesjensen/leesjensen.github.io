For

For loops can iterate arrays and objects as well as generate numbers.

  • [{{i}}] {{item}}
  • {{n}} 
      <div id="app">
        <ul>
          <li v-for="(item, i) in items">[{{i}}] {{item}}</li>
          <li><span v-for="n in 5" v-if="n != 3">{{n}} </span></li>
        </ul>

        <template v-for="(val, key) in obj">
          <label>{{ key }}</label>
          <span>➜ Created by {{val.by}} in {{val.date}}.</span>
          <br />
        </template>
      </div>

      <script>
        var app = new Vue({
          el: '#app',
          data: {
            items: ['Lee', 'Lie', 'Linus'],
            obj: {
              html: { by: 'Lee', date: '1993' },
              css: { by: 'Lie', date: '1996' },
              git: { by: 'Linus', date: '2005' },
            },
          },
        });
      </script>
