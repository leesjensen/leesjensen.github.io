Basic Component

Components modularize functionality that is represented by new HTML elements. 

<div id="app">
  <app-trooper></app-trooper>
  <app-trooper></app-trooper>
  <app-trooper></app-trooper>
</div>

<script>
  Vue.component('app-trooper', {
    template: '<div class="trooper">{{operatingNumber}}</div>',
    data: function () {
      return {
        operatingNumber: `TK-${Math.floor(Math.random() * 1000)}`,
      };
    },
  });

  var app = new Vue({
    el: '#app',
  });
</script>
