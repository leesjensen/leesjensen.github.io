Component Slots

Components can define content children as represented by template slots. There is always a default slot. Slots can be named and have default content. 

<div id="app">
  <app-layout>
    <template v-slot:header><div>Header</div></template>
    <template v-slot:sidenav><div>Side Navigation</div></template>
    <template v-slot:content><div>Content Body</div></template>
  </app-layout>
</div>

<script>
  Vue.component('app-layout', {
    template: `
      <section class='layout'>
        <div class="layout-header">
          <slot name="header"></slot>
        </div>
        <div class="layout-main">
          <div class="layout-sidenav">
            <slot name="sidenav"></slot>
          </div>
          <div class="layout-content">
            <slot name="content"></slot>
          </div>
        </div>
        <div class="layout-footer">
          <slot name="footer">Default footer content</slot>
        </div>
      </section>
    `,
  });

  var app = new Vue({
    el: '#app',
  });
</script>
