Component Properties

Components can have properties just like native HTML elements.

      <div id="app">
        <app-photo grayscale></app-photo>
        <app-photo :height="200"></app-photo>
        <app-photo :width="200" :grayscale="false" blur></app-photo>
      </div>

      <script>
        Vue.component('app-photo', {
          props: {
            height: { type: Number, default: 100 },
            width: { type: Number, default: 100 },
            grayscale: Boolean,
            blur: Boolean,
          },
          template: '<img v-bind:src="imageUrl" />',
          computed: {
            imageUrl: function () {
              const params = [];
              this.grayscale ? params.push('grayscale') : null;
              this.blur ? params.push('blur=3') : null;
              return `https://picsum.photos/seed/${Math.random()}/${
                this.width
              }/${this.height}?${params.join('&')}`;
            },
          },
        });

        var app = new Vue({
          el: '#app',
        });
      </script>
